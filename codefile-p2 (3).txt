  // working with a linked list : elements of short type 

  struct node 
  { unsigned short num;
    struct node * next;
  };
  // list of functions
  bool isempty ( struct node * list )
  { if ( list == 0 ) return true; else return false; }

  // length(list) - returns the number of elements in list
  int length ( struct node * list )
  { struct node * p = list;
    int count = 0;
    while ( p != 0 ) { count ++; p = p->next; };
    return count;
  }

  bool  find (struct node *list, unsigned short elm )
  { struct node *p = list;
    bool found = false;
    while ( p != 0 )
    { if (p->num == elm) {found = true; break;};
      p = p->next;
    };
    return (found);
  }

  void append (struct node ** list, unsigned short elm )
  { // find the last node of list; empty list is easy to handle
    struct node * p = (struct node*) malloc(sizeof(struct node));
     p->num = elm; p->next = 0;
      if ( isempty(*list)) *list = p;
      else
      {  struct node * curr = *list;
         struct node * prev = *list;
         while ( curr != 0 )
         {prev = curr; curr = curr->next;};
         prev->next = p;
      };
      return ;
  }

  void makelist(struct node ** list, short elm )
  { struct node *p = (struct node*) malloc (sizeof(struct node));
    if (elm == -1) {p = 0; *list = p;}
    else { p->num = elm; p->next = 0; *list = p;}
    return;
  }
